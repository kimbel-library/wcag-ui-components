1. Introduction

The DataTableExt module transforms HTML tables into fully interactive, AJAX-driven data interfaces. It supports sorting, searching, pagination, row selection, keyboard accessibility, and responsive resizing.

2. Component Overview

qsa('table.dataTable').forEach(table => {
    Object.assign(table, DataTableExt);
    table.init();
});

4. Initialization & Lifecycle

init(argsIn) {
    const url = table.getAttribute('data-url');
    const args = argsIn ? { ...dataArgs, ...argsIn } : { ...dataArgs };

    ajaxRequest(url, {
        method: 'GET',
        parameters: args,
        onSuccess: transport => {
            const result = transport.responseJSON || {};
            table.tableData = result.data || [];
            this.displayData();
            this.displayPagination();
            this.searchHook();
            this.sortHook();
            this.selectHook();
            this.resizeHook();
        }
    });
}

6. Rendering & Pagination

displayData(start = 0) {
    const tbody = qs('tbody.dataTableContent');
    tbody.innerHTML = '';

    table.tableData.slice(start).forEach(obj => {
        const tr = document.createElement('tr');
        Object.values(obj).forEach(val => {
            const td = document.createElement('td');
            td.textContent = val;
            tr.appendChild(td);
        });
        tbody.appendChild(tr);
    });
}

7. Row Selection System

delegate(table, 'tbody tr', 'click', (event, tr) => {
    if (!table.hasAttribute('select-multiple')) {
        this.clearSelection();
    }
    tr.classList.toggle('highlight');
});

8. Searching

input.addEventListener('keydown', event => {
    if (event.key === 'Enter') {
        const value = input.value;
        if (/^\s*$/.test(value)) table.refreshData();
        else table.refreshData({ search: value });
    }
});

9. Sorting

th.addEventListener('click', () => {
    const sortKey = th.dataset.sort;
    const descending = th.dataset.order === 'desc';

    table.tableData.sortElementsBy({
        prop: sortKey,
        desc: !descending
    });

    th.dataset.order = descending ? 'asc' : 'desc';
    this.displayData();
});

if (toggle.getAttribute('data-action') === 'highlight only') {

11. Resizing

status.addEventListener('mousedown', evt => {
    const onMove = e => {
        const newHeight = e.pageY - container.offsetTop;
        if (newHeight > 55) {
            container.style.height = newHeight + 'px';
        }
    };
    document.addEventListener('mousemove', onMove);
    document.addEventListener('mouseup', () => {
        document.removeEventListener('mousemove', onMove);
    });
});

12. Event Hooks

  • :dataLoaded — table finished rendering
  • :postSelect — row selection changed
  • :postSearch — search executed
